    class Solution(object):
    def longestPalindrome(self, s):
        def two_pointer(left, right):
            while left >= 0 and right < len(s) and s[left] == s[right]:
                left -= 1
                right += 1
            return s[left+1:right]

        if len(s) <= 1:
            return s

        result = ''
        for i in range(len(s)+1):
            result = max( result,
                        two_pointer(i, i+1),
                        two_pointer(i, i+2),
                        key = len)

        return result
