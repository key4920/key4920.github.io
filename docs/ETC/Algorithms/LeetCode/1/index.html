    class Solution(object):
    def twoSum(self, nums, target):
        for ind, num in enumerate(nums):
            sub_target = target - num
            if sub_target in nums[ind+1:]:
                sub_ind = nums.index(sub_target, ind+1, len(nums))
                return [ind, sub_ind]

    class Solution(object):
